function eK_b2bjewellers() {
  global $wpdb;
  $charset_collate = $wpdb->get_charset_collate();
  $table_name = $wpdb->prefix . 'b2bjewellers';
  $sql = "CREATE TABLE `$table_name` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `name` varchar(220) DEFAULT NULL,
  `price` varchar(220) DEFAULT NULL,
  `unit` varchar(220) DEFAULT NULL,
  PRIMARY KEY(id)
  ) ENGINE=MyISAM DEFAULT CHARSET=latin1;
  ";
  if ($wpdb->get_var("SHOW TABLES LIKE '$table_name'") != $table_name) {
    require_once(ABSPATH . 'wp-admin/includes/upgrade.php');
    dbDelta($sql);
  }
}

function return_custom_price1($product) {
    global $wpdb;
    $table_name1 = $wpdb->prefix . 'b2bjewellers';
    $query = "SELECT * FROM $table_name1";
    $results = $wpdb->get_results($query);
    $post_id = $product;
    $table_name = $wpdb->prefix . 'postmeta';
    $result = $wpdb->get_results("SELECT * FROM $table_name where post_id='$post_id'");   
    $myvals = get_post_meta($post_id);
    $someJSON = $myvals['_item_type'][0];
    $decoded = json_decode($someJSON);
    $mul=0;
    if(!empty($decoded)){
      foreach ($decoded as $key=>$arr){
        foreach ($results as $res){
          $user_val=$arr->id;
          $plug_val=$res->id;
          if($plug_val == $user_val){
            $table_update = $wpdb->prefix . 'postmeta';
            $update = $arr;
            $update->total = ($arr->qty * $res->price);
            $newResult = $decoded;
            $newResult[$key] = $update;
            $newResult = json_encode($newResult);
            $mul += ($arr->qty * $res->price);
            $price = $mul;
            $check_price = $wpdb->get_results("SELECT * FROM $table_name where post_id='$post_id' AND meta_key = '_price'");
            // var_dump($check);exit;
            if(empty($check_price)){
             $wpdb->query("INSERT into $table_update  (post_id,meta_key,meta_value) values ($post_id,'_price',$price) ");
             $wpdb->query("INSERT into $table_update  (post_id,meta_key,meta_value) values ($post_id,'_item_type',$newResult) ");
            }
           else{
            $wpdb->query("UPDATE $table_update SET meta_value='$price' WHERE post_id='$post_id' AND meta_key = '_price' ");
            $wpdb->query("UPDATE $table_update SET meta_value='$newResult' WHERE post_id='$post_id' AND meta_key = '_item_type' ");
           }
       
           $check_regular_price = $wpdb->get_results("SELECT * FROM $table_name where post_id='$post_id' AND meta_key = '_regular_price'");
           if(empty($check_regular_price))
             $wpdb->query("INSERT into $table_update  (post_id,meta_key,meta_value) values ($post_id,'_regular_price',$price) ");
           else
             $wpdb->query("UPDATE $table_update SET meta_value='$price' WHERE post_id='$post_id' AND meta_key = '_regular_price' ");
       
              }
          }
      }
  }
    
    return $mul;
